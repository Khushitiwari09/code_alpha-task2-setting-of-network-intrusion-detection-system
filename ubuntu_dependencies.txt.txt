snort_install.txt:

bash
# Install Snort on Ubuntu
sudo apt update
sudo apt install -y snort libpcap-dev libpcre3-dev libdumbnet-dev

# Verify installation
snort -V

# Initialize Snort
sudo snort -A console -q -u snort -g snort -c /etc/snort/snort.conf -i eth0
